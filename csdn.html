Problem Description
Calculate A + B.

Input
Each line will contain two integers A and B. Process to end of file.

Output
For each case, output A + B in one line.

Sample Input
1 1

Sample Output
2

Submission

#include <stdio.h>
int main()
{
    int a,b;
  // 在while循环中以EOF作为文件结束标志
  // ASCII代码值的范围是0~255，不可能出现-1，因此可以用EOF作为文件结束标志
    while(scanf("%d %d",&a,&b)!=EOF){
  //while (~scanf("%d%d", &x, &y))可以表示循环输入
  // 输出语句
    printf("%d\n",(a+b));
    }
    return 0;
}



#include <iostream>
using namespace std;

int main()
{
    int a,b;
    while(cin>>a>>b)
    {
        cout<<a+b<<endl;
    }
    return 0;
}
